Shell Scripts are made of commands with structure helpers

  • if/elif/else/fi

#!/bin/bash

FILENAME="$1"

if [ -f ${FILENAME} ]; then
  echo "${FILNAME} is a file"
elif [ -d ${FILENAME} ]; then
  echo "${FILENAME} is a directory"
else
  echo "${FILENAME} is unknown"
fi
  • for

for x in 1 2 3; do
  echo ${x}
done
